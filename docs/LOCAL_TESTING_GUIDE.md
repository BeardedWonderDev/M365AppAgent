## 4Ô∏è‚É£ Automated Test Script

### Create Comprehensive Test Script
```bash
# Save as test-all.sh in project root
cat > test-all.sh << 'EOF'
#!/bin/bash

echo "üß™ AI TenantShield - Comprehensive Local Testing"
echo "================================================"

# Colors
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
CYAN='\033[0;36m'
NC='\033[0m' # No Color

# Test results
TESTS_PASSED=0
TESTS_FAILED=0

# Function to run test
run_test() {
    local test_name=$1
    local test_command=$2
    
    echo -e "\n${CYAN}Testing: $test_name${NC}"
    if eval "$test_command" > /dev/null 2>&1; then
        echo -e "${GREEN}‚úÖ $test_name passed${NC}"
        ((TESTS_PASSED++))
    else
        echo -e "${RED}‚ùå $test_name failed${NC}"
        ((TESTS_FAILED++))
    fi
}

# Backend Tests
echo -e "\n${YELLOW}üîß Backend Tests${NC}"
echo "=================="

run_test "PowerShell Module Manifest" \
    "pwsh -c 'Test-ModuleManifest ./backend/Modules/M365AIAgent/M365AIAgent.psd1'"

run_test "PowerShell Module Import" \
    "pwsh -c 'Import-Module ./backend/Modules/M365AIAgent/M365AIAgent.psd1 -Force'"

run_test "PowerShell Syntax Check" \
    "pwsh -c 'Get-ChildItem -Path ./backend -Recurse -Filter *.ps1 | ForEach-Object { \$null = [System.Management.Automation.PSParser]::Tokenize((Get-Content \$_.FullName -Raw), [ref]\$null) }'"

# Mobile Tests
echo -e "\n${YELLOW}üì± Mobile Tests${NC}"
echo "==============="

if command -v swift &> /dev/null; then
    run_test "Swift Build" \
        "cd mobile && swift build"
    
    run_test "Swift Tests" \
        "cd mobile && swift test"
else
    echo -e "${YELLOW}‚ö†Ô∏è  Swift not installed, skipping mobile tests${NC}"
fi

# Function App Tests
echo -e "\n${YELLOW}‚ö° Azure Functions Tests${NC}"
echo "========================"

if command -v func &> /dev/null; then
    run_test "Function App Validation" \
        "cd backend && func host start --dry-run"
else
    echo -e "${YELLOW}‚ö†Ô∏è  Azure Functions Core Tools not installed${NC}"
fi

# Summary
echo -e "\n${CYAN}================================${NC}"
echo -e "${CYAN}Test Summary${NC}"
echo -e "${CYAN}================================${NC}"
echo -e "${GREEN}Passed: $TESTS_PASSED${NC}"
echo -e "${RED}Failed: $TESTS_FAILED${NC}"

if [ $TESTS_FAILED -eq 0 ]; then
    echo -e "\n${GREEN}üéâ All tests passed!${NC}"
    exit 0
else
    echo -e "\n${RED}‚ö†Ô∏è  Some tests failed. Please review the output above.${NC}"
    exit 1
fi
EOF

chmod +x test-all.sh

# Run the test script
./test-all.sh
```

## üîç Troubleshooting

### Common Issues and Solutions

1. **Module Import Fails**
   ```powershell
   # Check PowerShell version
   $PSVersionTable.PSVersion
   # Should be 7.4 or higher
   ```

2. **Swift Build Fails**
   ```bash
   # Ensure Skip is installed
   skip checkup
   # Update Skip
   brew upgrade skiptools/skip/skip
   ```

3. **Functions Won't Start**
   ```bash
   # Check Node.js version
   node --version
   # Should be 18.x or higher
   
   # Reinstall Azure Functions Core Tools
   brew reinstall azure-functions-core-tools@4
   ```

4. **Tests Timeout**
   ```powershell
   # Increase timeout in test scripts
   $PSDefaultParameterValues['Invoke-Pester:Timeout'] = 120
   ```
